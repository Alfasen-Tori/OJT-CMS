[USERS]
id (PK, auto-increment)
email (unique)
password
fname
lname
contact
timestamps


[ADMINS]
id (PK, auto-increment)
faculty_id (VARCHAR) — unique
user_id (FK to users.id)
timestamps
	
[COORDINATORS]
id (PK, auto-increment)
faculty_id (VARCHAR, e.g. "2013-07896")
user_id (FK to users.id)
dept_id (FK to departments.dept_id)
can_add_hte (Enum:0 = true, 1 = false)
timestamps

[INTERNS]
id (PK, auto-increment)
student_id (VARCHAR, e.g. "2022-09709") — unique
user_id (FK to users.id)
dept_id (FK to departments.dept_id)
coordinator_id (FK from coordinator.id)
birthdate (date)
section (ENUM: 'a', 'b', 'c', 'd', 'e', 'f')
year_level (TINYINT, 1–4)
academic_year (e.g. "2025–2026")
semester (ENUM: "1st", "2nd", "Midyear")
status (ENUM: "pending requirements", "ready for deployment", "endorsed", "processing", "deployed") default is pending requirements for newly added interns
first_login (tiny int) default = 1
timestamps

[DEPARTMENTS]
dept_id (PK, auto-increment)
dept_name (e.g. "Information Technology")
short_name (e.g. "BSIT")
college_id (pk)
timestamps

[COLLEGES]
id (pk)
name
short_name
timstamps

[HTES]
id (PK, auto-increment)
user_id (FK to users.id)
organization_name
status (ENUM: "active", "new")
type (ENUM: 'private', 'government', 'NGO', 'Educational', 'Other')
address (TEXT)
description (TEXT)
slots (INT)
moa_path (VARCHAR or TEXT)
moa_is_signed (Enum: "no", "yes") // default is no
first_login (tinyint) default = 1
timestamps	

[SKILLS]
skill_id
dept_id (FK)
name (eg:  web development)

[student_skill] 
id (PRI)
intern_id (FK)
skills_id (FK)
timestamps

[hte_skill]
id (PRI)
hte_id (FK)
skill_id (FK)
timestamps

[INTERNS_HTE]
id (PK, auto-increment)
intern_id (FK from interns.id)
hte_id (FK to htes.id)
coordinator_id
status (ENUM: "endorsed", "deployed") — default: "endorsed"
start_date,
end_date,
no_of_hours, - amount of hours interns are required to render
no_of_weeks, 
endorsed_at (timestamp) — when the endorsement happened
deployed_at (timestamp, nullable) — when deployment happened
coordinator_id (fk from coordinator id) - this way we can both access who deployed the students to the HTE as well as the department 
endorsement_letter_path
timestamps (created_at, updated_at)

[ATTENDANCES]
id (PK, auto-increment)
intern_hte_id (FK to interns_hte.id)
date (DATE)
time_in (DATETIME, nullable)
time_out (DATETIME, nullable)
hours_rendered (DECIMAL(5,2), nullable) — will be used to track progress. internship will be completed when this >= no_of_hours in interns_hte
timestamps

[WEEKLY_REPORTS]
id (PK, auto-increment)
intern_id (FK to interns.id)
week_no (TINYINT) — e.g., 1, 2, 3...
report_path (VARCHAR or TEXT) — uploaded file path, nullable
submitted_at (TIMESTAMP) nullable
timestamps

[INTERN_EVALUATIONS]
id (PK, auto-increment)
intern_hte_id (FK to interns_hte.id) - to track which deployment is being evaluated
grade (DECIMAL(5,2) - 0-100 scale
comments (TEXT, nullable) - optional feedback from HTE
evaluation_date (DATE)
timestamps (created_at, updated_at)
